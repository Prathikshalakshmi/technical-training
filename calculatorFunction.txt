{
    //function proto-type
    double *calc(double, double, char);
    double *result = calc(10.5, 0, '/');
    if( result !=NULL)
       printf("%lf", *result);
    else
        printf("Invalid operation/Denominator should not be zero");

    return 0;
}

//function to implementsimple calculator
double* calc(double num1, double num2, char opr){
    double *result = (double*)malloc(sizeof(double));
    switch(opr){
        case '+': *result = (num1 + num2); break;
        case '-': *result = (num1 - num2); break;
        case '*': case 'x': case 'X': *result = (num1 * num2); break;
        case '/': 
                if(num2 != 0)
                    *result = (num1 / num2);
                else 
                    return NULL;
                break;
        case '%': *result = fmod(num1, num2); break; 
        default: return NULL;
    }
    return result;
}
