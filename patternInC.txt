//Program to print the following pattern
/*
input 
5
output
1
12
123
1234
12345

*/
#include <stdio.h>

int main(){
    int n;
    scanf("%d", &n);
    for (int r = 1;r<=n; r++){
        for (int c = 1; c<=r; c++){
            printf("%d", c);
        }
        printf("\n");
    }
        
    return 0;
}



//Program to print the following pattern
/*
input 
5
output
54321
 4321
  321
   21
    1
*/
#include <stdio.h>
int main(){
    int n;
    scanf("%d", &n);
    for (int r = n;r>=1; r--){
        //printing space
        for (int c = 1;c<=n-r;c++ )
            printf(" ");
        for (int c = r; c>=1; c--){
            printf("%d", c);
        }
        printf("\n");
    }
    return 0;
}
