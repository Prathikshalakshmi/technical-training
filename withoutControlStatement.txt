//finding maximum value without control statement
#include<stdlib.h>
#include <stdio.h>
int main()
{
   int x=10,y=20;
   
   printf("MAX : %d ",((x+y)+abs(x-y))/2);
   
   return 0;
}


//finding minimum value without control statement
#include<stdlib.h>
#include <stdio.h>
int main()
{
   int x=10,y=20;
   
   printf("MIN : %d ",((x+y)-abs(x-y))/2);
   
   return 0;
}
